function doGet(e) {
  if (!e || !e.parameter) {
    return respond({ ok: false, error: "No request parameters received." }, null);
  }

  const p = e.parameter;
  const key = p.key;
  const cb = p.callback;
  if (key !== SECRET_KEY) return respond({ error: "Unauthorized" }, cb);

  const action = p.action || "loadApplicants";
  try {
    if (action === "loadApplicants" || action === "load") {
      return respond(loadApplicants_(p.sheet || "Sheet1"), cb);
    }
    if (action === "screen") {
      return respond(screenWithGemini_(p.prompt || ""), cb);
    }
    if (action === "screenApplicant") {
      return respond(screenApplicant_(p.sheet || "Sheet1", p.applicationId || ""), cb);
    }
    if (action === "saveAssignment" || action === "assign") {
      return respond(saveAssignment_({
        applicationId: p.applicationId || p.ventureId || "",
        reviewer1: p.reviewer1 || "",
        reviewer2: p.reviewer2 || ""
      }), cb);
    }
    if (action === "saveAiDraft") {
      return respond(saveAiDraft_({
        applicationId: p.applicationId || "",
        ventureName: p.ventureName || "",
        applicantName: p.applicantName || "",
        draft: safeJsonParse_(p.draft, {})
      }), cb);
    }
    if (action === "saveReview" || action === "save") {
      return respond(saveReview_(parseReviewPayload_(p)), cb);
    }
    if (action === "saveResolution") {
      return respond(saveResolution_({
        applicationId: p.applicationId || "",
        resolutionState: p.resolutionState || "required",
        resolutionVerdict: p.resolutionVerdict || "",
        resolutionNotes: p.resolutionNotes || "",
        resolvedBy: p.resolvedBy || "",
        resolvedAt: p.resolvedAt || "",
        reviewerPair: p.reviewerPair || ""
      }), cb);
    }
    return respond({ error: "Unknown action" }, cb);
  } catch (err) {
    return respond({ error: err.toString() }, cb);
  }
}

function doPost(e) {
  try {
    const payload = safeJsonParse_(e.postData && e.postData.contents, {});
    if (payload.key !== SECRET_KEY) {
      return respond({ error: "Unauthorized" });
    }

    if (payload.action === "loadApplicants" || payload.action === "load" || !payload.action) {
      return respond(loadApplicants_(payload.sheet || "Sheet1"));
    }
    if (payload.action === "screen") {
      return respond(screenWithGemini_(payload.prompt || ""));
    }
    if (payload.action === "screenApplicant") {
      return respond(screenApplicant_(payload.sheet || "Sheet1", payload.applicationId || ""));
    }
    if (payload.action === "saveAssignment" || payload.action === "assign") {
      return respond(saveAssignment_({
        applicationId: payload.applicationId || "",
        reviewer1: payload.reviewer1 || "",
        reviewer2: payload.reviewer2 || ""
      }));
    }
    if (payload.action === "saveAiDraft") {
      return respond(saveAiDraft_({
        applicationId: payload.applicationId || "",
        ventureName: payload.ventureName || "",
        applicantName: payload.applicantName || "",
        draft: payload.draft || {}
      }));
    }
    if (payload.action === "saveReview" || payload.action === "save") {
      return respond(saveReview_(parseReviewPayload_(payload)));
    }
    if (payload.action === "saveResolution") {
      return respond(saveResolution_({
        applicationId: payload.applicationId || "",
        resolutionState: payload.resolutionState || "required",
        resolutionVerdict: payload.resolutionVerdict || "",
        resolutionNotes: payload.resolutionNotes || "",
        resolvedBy: payload.resolvedBy || "",
        resolvedAt: payload.resolvedAt || "",
        reviewerPair: payload.reviewerPair || ""
      }));
    }

    return respond({ error: "Unknown action" });
  } catch (err) {
    return respond({ error: err.toString() });
  }
}

function createGeminiDraftSchema_() {
  const gateProperties = {};
  const scoreProperties = {};
  const commentProperties = {};

  AI_GATE_KEYS.forEach(key => {
    gateProperties[key] = {
      type: "object",
      propertyOrdering: ["verdict", "note"],
      additionalProperties: false,
      properties: {
        verdict: { type: "string", enum: ["pass", "fail", "flag"] },
        note: { type: "string" }
      },
      required: ["verdict", "note"]
    };
  });

  AI_SCORE_KEYS.forEach(key => {
    scoreProperties[key] = { type: "integer", minimum: 1, maximum: 5 };
    commentProperties[key] = { type: "string" };
  });

  return {
    type: "object",
    propertyOrdering: ["eligibility", "scores", "category_comments", "summary"],
    additionalProperties: false,
    properties: {
      eligibility: {
        type: "object",
        propertyOrdering: AI_GATE_KEYS.slice(),
        additionalProperties: false,
        properties: gateProperties,
        required: AI_GATE_KEYS.slice()
      },
      scores: {
        type: "object",
        propertyOrdering: AI_SCORE_KEYS.slice(),
        additionalProperties: false,
        properties: scoreProperties,
        required: AI_SCORE_KEYS.slice()
      },
      category_comments: {
        type: "object",
        propertyOrdering: AI_SCORE_KEYS.slice(),
        additionalProperties: false,
        properties: commentProperties,
        required: AI_SCORE_KEYS.slice()
      },
      summary: { type: "string" }
    },
    required: ["eligibility", "scores", "category_comments", "summary"]
  };
}

function buildGeminiRequestPayload_(prompt) {
  return {
    contents: [{ parts: [{ text: prompt || "" }] }],
    generationConfig: {
      temperature: 0.1,
      maxOutputTokens: 2048,
      responseMimeType: "application/json",
      responseJsonSchema: GEMINI_DRAFT_SCHEMA
    }
  };
}

function extractGeminiJsonText_(data) {
  const raw = data.candidates &&
    data.candidates[0] &&
    data.candidates[0].content &&
    data.candidates[0].content.parts &&
    data.candidates[0].content.parts[0] &&
    data.candidates[0].content.parts[0].text
      ? data.candidates[0].content.parts[0].text
      : "";

  const cleaned = String(raw || "").replace(/```json|```/g, "").trim();
  const start = cleaned.indexOf("{");
  const end = cleaned.lastIndexOf("}");
  if (start === -1 || end === -1) throw new Error("Gemini response did not contain a JSON object.");
  return cleaned.slice(start, end + 1).replace(/[\x00-\x1F\x7F]/g, " ");
}

function getApplicantRows_(sheetName) {
  const ss = SpreadsheetApp.openById(SPREADSHEET_ID);
  const sheet = ss.getSheetByName(sheetName);
  if (!sheet) throw new Error("Sheet '" + sheetName + "' not found");

  const raw = sheet.getDataRange().getValues();
  if (raw.length < 4) return [];

  const headers = raw[2].map(v => String(v).trim());
  return raw.slice(3)
    .filter(r => r.some(c => c !== "" && c !== null && c !== undefined))
    .map(r => {
      const obj = {};
      headers.forEach((header, idx) => {
        obj[header] = r[idx] !== null && r[idx] !== undefined ? String(r[idx]).trim() : "";
      });
      return obj;
    });
}

function findApplicantRow_(sheetName, applicationId) {
  const targetId = String(applicationId || "").toLowerCase().trim();
  if (!targetId) throw new Error("Application ID is required");

  const rows = getApplicantRows_(sheetName);
  for (let i = 0; i < rows.length; i++) {
    if (applicantIdFromRow_(rows[i]) === targetId) return rows[i];
  }
  throw new Error("Applicant '" + applicationId + "' was not found in sheet '" + sheetName + "'");
}

function screenApplicant_(sheetName, applicationId) {
  const row = findApplicantRow_(sheetName, applicationId);
  return screenWithGemini_(buildApplicantPrompt_(row));
}

  return {
    ok: true,
    rows: rows,
    assignments: getAssignments_(),
    savedReviews: getSavedReviews_(),
    resolutions: getResolutions_(),
    aiDrafts: getAiDrafts_(),
    total: rows.length
  };
}

function saveAiDraft_(d) {
  if (!d.applicationId) throw new Error("Application ID is required");
  const sh = ensureSheet_(AI_DRAFTS_SHEET, AI_DRAFT_HEADERS);
  const data = sh.getDataRange().getValues();
  const row = [
    d.applicationId,
    new Date().toISOString(),
    d.ventureName || "",
    d.applicantName || "",
    JSON.stringify(d.draft || {})
  ];

  for (let i = 1; i < data.length; i++) {
    if (String(data[i][0] || "").toLowerCase().trim() === d.applicationId.toLowerCase().trim()) {
      sh.getRange(i + 1, 1, 1, row.length).setValues([row]);
      return { ok: true, mode: "updated" };
    }
  }

  sh.appendRow(row);
  return { ok: true, mode: "created" };
}

function screenWithGemini_(prompt) {
  try {
    if (!GEMINI_KEY) {
      return { ok: false, error: "Gemini API key is missing in Apps Script. Set GEMINI_KEY or use browser-side Gemini." };
    }

    const errors = [];

    for (let m = 0; m < GEMINI_MODELS.length; m++) {
      const model = GEMINI_MODELS[m];
      for (let attempt = 0; attempt < 2; attempt++) {
        try {
          const response = UrlFetchApp.fetch(
            "https://generativelanguage.googleapis.com/v1beta/models/" + encodeURIComponent(model) + ":generateContent?key=" + encodeURIComponent(GEMINI_KEY),
            {
              method: "post",
              contentType: "application/json",
              payload: JSON.stringify(buildGeminiRequestPayload_(prompt || "")),
              muteHttpExceptions: true
            }
          );

          const data = safeJsonParse_(response.getContentText(), null);
          if (!data) throw new Error("Gemini returned invalid JSON.");
          if (data.error) {
            const code = Number(data.error.code || response.getResponseCode() || 0);
            const msg = data.error.message || JSON.stringify(data.error);
            if (code === 429 || code === 503) throw new Error(msg);
            throw new Error(msg);
          }

          const parsed = safeJsonParse_(extractGeminiJsonText_(data), null);
          if (!parsed) throw new Error("Gemini JSON could not be parsed.");

          return { ok: true, result: parsed };
        } catch (err) {
          errors.push(model + " attempt " + (attempt + 1) + ": " + err.toString());
          if (attempt < 1) Utilities.sleep((attempt + 1) * 2000);
        }
      }
    }

    return { ok: false, error: errors.join(" | ") || "Gemini request failed." };
  } catch (err) {
    return { ok: false, error: err.toString() };
  }
}

function ensureSheet_(name, headers) {
  const ss = SpreadsheetApp.openById(SPREADSHEET_ID);
  let sh = ss.getSheetByName(name);
  if (!sh) {
    sh = ss.insertSheet(name);
    sh.getRange(1, 1, 1, headers.length).setValues([headers]);
    styleHeader_(sh, headers.length);
    return sh;
  }

  const current = sh.getRange(1, 1, 1, Math.max(sh.getLastColumn(), 1)).getValues()[0].map(String);
  let changed = false;
  headers.forEach((header, idx) => {
    if (current[idx] !== header) {
      sh.getRange(1, idx + 1).setValue(header);
      changed = true;
    }
  });
  if (changed) styleHeader_(sh, headers.length);
  if (sh.getFrozenRows() !== 1) sh.setFrozenRows(1);
  return sh;
}

function styleHeader_(sheet, count) {
  sheet.getRange(1, 1, 1, count)
    .setBackground("#1a2e5a")
    .setFontColor("#ffffff")
    .setFontWeight("bold");
  sheet.setFrozenRows(1);
}

function getAssignments_() {
  const ss = SpreadsheetApp.openById(SPREADSHEET_ID);
  const sh = ss.getSheetByName(ASSIGNMENTS_SHEET);
  if (!sh || sh.getLastRow() < 2) return {};
  const values = sh.getRange(2, 1, sh.getLastRow() - 1, Math.max(sh.getLastColumn(), 4)).getValues();
  const map = {};
  values.forEach(row => {
    const applicationId = String(row[0] || "").toLowerCase().trim();
    if (!applicationId) return;
    map[applicationId] = {
      reviewer1: String(row[1] || "").trim(),
      reviewer2: String(row[2] || "").trim(),
      updatedAt: String(row[3] || "").trim()
    };
  });
  return map;
}

function getSavedReviews_() {
  const ss = SpreadsheetApp.openById(SPREADSHEET_ID);
  const sh = ss.getSheetByName(RESULTS_SHEET);
  if (!sh || sh.getLastRow() < 2) return {};

  const headers = sh.getRange(1, 1, 1, sh.getLastColumn()).getValues()[0].map(String);
  const idx = indexMap_(headers);
  if (idx["Review Key"] === undefined || idx["Application ID"] === undefined || idx["Reviewer"] === undefined) {
    return {};
  }

  const values = sh.getRange(2, 1, sh.getLastRow() - 1, sh.getLastColumn()).getValues();
  const out = {};
  values.forEach(row => {
    const applicationId = String(row[idx["Application ID"]] || "").toLowerCase().trim();
    const reviewer = String(row[idx["Reviewer"]] || "").trim();
    if (!applicationId || !reviewer) return;

    if (!out[applicationId]) out[applicationId] = {};
    const eligibility = safeJsonParse_(row[idx["Eligibility JSON"]], {});
    out[applicationId][reviewer] = {
      reviewer: reviewer,
      reviewerSlot: idx["Reviewer Slot"] !== undefined ? String(row[idx["Reviewer Slot"]] || "").trim() : "",
      gates: eligibility.gates || {},
      eligibility: eligibility,
      ratings: safeJsonParse_(row[idx["Ratings JSON"]], {}),
      categoryComments: safeJsonParse_(row[idx["Category Comments JSON"]], {}),
      editedGates: splitList_(idx["Edited Gates"] !== undefined ? row[idx["Edited Gates"]] : ""),
      editedRatings: splitList_(idx["Edited Ratings"] !== undefined ? row[idx["Edited Ratings"]] : ""),

function getResolutions_() {
  const ss = SpreadsheetApp.openById(SPREADSHEET_ID);
  const sh = ss.getSheetByName(RESOLUTIONS_SHEET);
  if (!sh || sh.getLastRow() < 2) return {};
  const values = sh.getRange(2, 1, sh.getLastRow() - 1, Math.max(sh.getLastColumn(), 7)).getValues();
  const out = {};
  values.forEach(row => {
    const applicationId = String(row[0] || "").toLowerCase().trim();
    if (!applicationId) return;
    out[applicationId] = {
      resolutionState: String(row[1] || "").trim() || "required",
      resolutionVerdict: mapVerdictLabel_(String(row[2] || "").trim()),
      resolutionNotes: String(row[3] || ""),
      resolvedBy: String(row[4] || "").trim(),
      resolvedAt: String(row[5] || "").trim(),
      reviewerPair: String(row[6] || "").trim()
    };
  });
  return out;
}

function splitList_(value) {
  const str = String(value || "").trim();
  return str ? str.split(/\s*,\s*/) : [];
}

function safeJsonParse_(value, fallback) {
  try {
    if (value === null || value === undefined || value === "") return fallback;
    return JSON.parse(String(value));
  } catch (err) {
    return fallback;
  }
}

function indexMap_(headers) {
  const out = {};
  headers.forEach((header, idx) => out[String(header || "").trim()] = idx);
  return out;
}
